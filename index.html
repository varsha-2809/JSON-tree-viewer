JSON Tree Viewer

document.getElementById('format-btn').addEventListener('click', () => { const input = document.getElementById('json-input').value; const container = document.getElementById('jsoneditor'); const options = { mode: 'tree', modes: ['view'], // allowed modes onError: function (err) { alert(err.toString()); } }; const editor = new JSONEditor(container, options); try { const json = JSON.parse(input); editor.set(json); } catch (error) { alert('Invalid JSON'); } });
